using Microsoft.EntityFrameworkCore;
using pgDataAccess.Models;

namespace pgDataAccess.Services;

/// <summary>
/// Сервис для управления справочниками: категории, производители, поставщики
/// Доступно только администраторам
/// </summary>
public class ReferenceService
{
    /// <summary>
    /// Получить все категории
    /// </summary>
    public async Task<List<Category>> GetCategoriesAsync()
    {
        using var db = new ApplicationDbContext();
        return await db.Categories.OrderBy(c => c.Name).ToListAsync();
    }

    /// <summary>
    /// Добавить новую категорию
    /// </summary>
    public async Task<Category> AddCategoryAsync(string name)
    {
        using var db = new ApplicationDbContext();
        var category = new Category { Name = name };
        db.Categories.Add(category);
        await db.SaveChangesAsync();
        return category;
    }

    /// <summary>
    /// Обновить категорию
    /// </summary>
    public async Task UpdateCategoryAsync(Category category)
    {
        using var db = new ApplicationDbContext();
        db.Categories.Update(category);
        await db.SaveChangesAsync();
    }

    /// <summary>
    /// Удалить категорию
    /// </summary>
    public async Task DeleteCategoryAsync(int id)
    {
        using var db = new ApplicationDbContext();
        var category = await db.Categories.FindAsync(id);
        if (category != null)
        {
            db.Categories.Remove(category);
            await db.SaveChangesAsync();
        }
    }

    /// <summary>
    /// Получить все категории (для комбобокса)
    /// </summary>
    public async Task<List<Category>> GetCategoriesForComboBoxAsync()
    {
        return await GetCategoriesAsync();
    }

    /// <summary>
    /// Получить все производства
    /// </summary>
    public async Task<List<Manufacturer>> GetManufacturersAsync()
    {
        using var db = new ApplicationDbContext();
        return await db.Manufacturers.OrderBy(m => m.Name).ToListAsync();
    }

    /// <summary>
    /// Добавить нового производителя
    /// </summary>
    public async Task<Manufacturer> AddManufacturerAsync(string name)
    {
        using var db = new ApplicationDbContext();
        var manufacturer = new Manufacturer { Name = name };
        db.Manufacturers.Add(manufacturer);
        await db.SaveChangesAsync();
        return manufacturer;
    }

    /// <summary>
    /// Обновить производителя
    /// </summary>
    public async Task UpdateManufacturerAsync(Manufacturer manufacturer)
    {
        using var db = new ApplicationDbContext();
        db.Manufacturers.Update(manufacturer);
        await db.SaveChangesAsync();
    }

    /// <summary>
    /// Удалить производителя
    /// </summary>
    public async Task DeleteManufacturerAsync(int id)
    {
        using var db = new ApplicationDbContext();
        var manufacturer = await db.Manufacturers.FindAsync(id);
        if (manufacturer != null)
        {
            db.Manufacturers.Remove(manufacturer);
            await db.SaveChangesAsync();
        }
    }

    /// <summary>
    /// Получить все поставки
    /// </summary>
    public async Task<List<Supplier>> GetSuppliersAsync()
    {
        using var db = new ApplicationDbContext();
        return await db.Suppliers.OrderBy(s => s.Name).ToListAsync();
    }

    /// <summary>
    /// Добавить нового поставщика
    /// </summary>
    public async Task<Supplier> AddSupplierAsync(string name)
    {
        using var db = new ApplicationDbContext();
        var supplier = new Supplier { Name = name };
        db.Suppliers.Add(supplier);
        await db.SaveChangesAsync();
        return supplier;
    }

    /// <summary>
    /// Обновить поставщика
    /// </summary>
    public async Task UpdateSupplierAsync(Supplier supplier)
    {
        using var db = new ApplicationDbContext();
        db.Suppliers.Update(supplier);
        await db.SaveChangesAsync();
    }

    /// <summary>
    /// Удалить поставщика
    /// </summary>
    public async Task DeleteSupplierAsync(int id)
    {
        using var db = new ApplicationDbContext();
        var supplier = await db.Suppliers.FindAsync(id);
        if (supplier != null)
        {
            db.Suppliers.Remove(supplier);
            await db.SaveChangesAsync();
        }
    }
}
